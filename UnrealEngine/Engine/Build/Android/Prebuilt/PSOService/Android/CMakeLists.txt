	add_definitions(-DNDEBUG)

    ${CMAKE_SOURCE_DIR}/../Source

    ${SOURCE_ROOT}/src

    ${SOURCE_ROOT}/PSOService.cpp
)

                     -DVK_USE_PLATFORM_ANDROID_KHR")

add_library(psoservice SHARED ${libpsoservice_source})

find_library(log-lib log )
